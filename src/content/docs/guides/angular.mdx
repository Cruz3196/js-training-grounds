This tutorial assumes you're already familiar with HTML, CSS, JS, and npm. It also assumes you know what a JS framework is for.

## Setup

First, visit the <a href="https://github.com/simpledevio/angular-training-ground" target="_blank">Angular training ground repo</a>. Press the green **Code** button and click **Download ZIP**. Unzip the file and then open it in your editor. Then, run this command in the terminal to install the packages listed in `package.json`.

```bash
npm install
```

Then run this command to start the dev server.

```bash
npm start
```

You should see something like this in your terminal.

```bash
Watch mode enabled. Watching for file changes...
NOTE: Raw file sizes do not reflect development server per-request transformations.
 ➜ Local: http://localhost:4200/
 ➜ press h + enter to show help
```

Visit <a href="http://localhost:4200/" target="_blank">localhost:4200</a> in your browser. The web page should say **Hello World**.

In your editor, you'll see several files. However, for now you'll mainly be working in `app.html`.

### Stackblitz

If you prefer to work online, you can also open the <a href="https://stackblitz.com/github/simpledevio/angular-training-ground?file=src/app/app.html" target="_blank">Angular training ground repo on Stackblitz</a> and start coding right away in your browser. You don't have to run any of the npm commands if you use Stackblitz.

### Filter items

```html title="todo-list.html" {10-20, 24-28}

    name="newTask"

      type="checkbox"
      [(ngModel)]="hideCompleted"
      name="hideCompleted"
    >
    Hide completed items
  </label>
</div>

<ul>
  @for (task of visibleTasks(); track $index) {
    <li>
        <input
          type="checkbox"
          [checked]="task.completed"
          (change)="toggleTask($index)"
        >
      {{ task.text }}
      <button (click)="deleteTask($index)">Delete</button>
    </li>
  }
</ul>
</div>
```

```ts title="todo-list.ts" {23-30, 45-54}
import { Component, signal, computed } from '@angular/core';
import { FormsModule } from '@angular/forms';

interface TodoItem {
  text: string;
  completed: boolean;
}

@Component({
  selector: 'app-todo-list',
  standalone: true,
  imports: [FormsModule],
  templateUrl: './todo-list.html',
})
export class TodoList {
  tasks = signal<TodoItem[]>([
    { text: 'Learn Angular basics', completed: false },
    { text: 'Build a todo app', completed: false },
    { text: 'Practice TypeScript', completed: false }
  ]);

  newTask = signal('');
  hideCompleted = signal(false);

  visibleTasks = computed(() => {
    const allTasks = this.tasks();
    return this.hideCompleted()
      ? allTasks.filter(task => !task.completed)
      : allTasks;
  });

  addTask() {
    const taskText = this.newTask().trim();
    if (taskText) {
      this.tasks.update(tasks => [...tasks, { text: taskText, completed: false }]);
      this.newTask.set('');
    }
  }

    deleteTask(index: number) {
    const taskToDelete = this.visibleTasks()[index];
    this.tasks.update(tasks => tasks.filter(task => task !== taskToDelete));
  }

    toggleTask(index: number) {
    const taskToToggle = this.visibleTasks()[index];
    this.tasks.update(tasks =>
      tasks.map(task =>
        task === taskToToggle
          ? { ...task, completed: !task.completed }
          : task
      )
    );
  }
}
```
